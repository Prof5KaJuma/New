import curses import time def text_animation(stdscr, text, delay=0.1): curses.curs_set(0) # Hide the cursor stdscr.clear() rows, cols = stdscr.getmaxyx() x = cols // 2 - len(text) // 2 y = rows // 2 for i in range(len(text) + 1): stdscr.addstr(y, x, text[:i]) stdscr.refresh() time.sleep(delay) stdscr.getch() # Wait for a key press def main(): text = "Incoming message..." curses.wrapper(text_animation, text) if __name__ == "__main__": main()
